function openProductDetails(product) {
activeProduct = product;

document.getElementById("modalTitle").innerText = product.title;
document.getElementById("modalPrice").innerText = `Rs. ${product.price}`;
document.getElementById("modalOldPrice").innerText = `Rs. ${product.oldPrice}`;
document.getElementById("modalDiscount").innerText = product.discount;
document.getElementById("modalDesc").innerText = product.description;

// Render Dynamic Overview or Default Text if Empty
document.getElementById("modalOverview").innerText = product.overview || "No overview details listed for this item.";

const sliderBox = document.getElementById("modalSliderBox");
sliderBox.innerHTML = "";

product.images.forEach((imgUrl) => {
const slide = document.createElement("div");
slide.className = "modal-slide-item";
slide.innerHTML = `<img src="${imgUrl}" alt="">`;
sliderBox.appendChild(slide);
});

document.getElementById("imageIndexLabel").innerText = `1 / ${product.images.length}`;

const thumbGallery = document.getElementById("modalThumbGallery");
thumbGallery.innerHTML = "";

if(product.images.length > 1) {
thumbGallery.style.display = "flex";
product.images.forEach((imgUrl, idx) => {
const thumb = document.createElement("div");
thumb.className = `thumb-item ${idx === 0 ? 'active' : ''}`;
thumb.id = `modal-thumb-${idx}`;
thumb.innerHTML = `<img src="${imgUrl}" alt="">`;

thumb.onclick = function() {
const width = sliderBox.clientWidth;
sliderBox.scrollTo({ left: width * idx, behavior: 'smooth' });
updateActiveThumbnail(idx);
};
thumbGallery.appendChild(thumb);
});
} else {
thumbGallery.style.display = "none";
}

sliderBox.onscroll = function() {
const width = sliderBox.clientWidth;
if(width === 0) return;
const activeIndex = Math.round(sliderBox.scrollLeft / width);
document.getElementById("imageIndexLabel").innerText = `${activeIndex + 1} / ${product.images.length}`;
updateActiveThumbnail(activeIndex);
};

if(localStorage.getItem("hamviro_pro_name")) {
document.getElementById("custName").value = localStorage.getItem("hamviro_pro_name");
document.getElementById("custPhone").value = localStorage.getItem("hamviro_pro_phone");
document.getElementById("custProvince").value = localStorage.getItem("hamviro_pro_province");
document.getElementById("custCity").value = localStorage.getItem("hamviro_pro_city");
document.getElementById("custAddress").value = localStorage.getItem("hamviro_pro_address");
selectAddressLabel(localStorage.getItem("hamviro_pro_label") || "Home");
}

document.getElementById("productModal").style.display = "flex";
document.body.style.overflow = "hidden";

setTimeout(() => { sliderBox.scrollLeft = 0; }, 50);
}

function filterCategory(catId) {
selectedCategory = catId; renderCategories();
document.getElementById("sectionTitle").innerText = catId === "all" ? "All Products" : `${catId.charAt(0).toUpperCase() + catId.slice(1).replace('-', ' ')} Deals`;

function runCountdown() {
let hours = 2, minutes = 14, seconds = 20;
setInterval(() => {
seconds--;
if(seconds < 0) { seconds = 59; minutes--; }
if(minutes < 0) { minutes = 59; hours--; }
document.getElementById("liveTimer").innerText = `${hours.toString().padStart(2, '0')} : ${minutes.toString().padStart(2, '0')} : ${seconds.toString().padStart(2, '0')}`;
}, 1000);
}
